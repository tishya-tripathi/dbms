1.	book, book_authors, BOOK_COPIES, library_branch

SELECT 	B.BOOK_ID, B.TITLE, B.PUBLISHER_NAME, 
	BA.AUTHOR_NAME, BC.NO_OF_COPIES, L.BRANCH_ID
FROM BOOK B, BOOK_AUTHORS BA, BOOK_COPIES BC, LIBRARY_BRANCH L
WHERE 	B.BOOK_ID=BA.BOOK_ID AND
	B.BOOK_ID=BC.BOOK_ID AND
	BC.BRANCH_ID=L.BRANCH_ID;

SELECT  CARD_NO
FROM 	BOOK_LENDING 
WHERE 	DATE_OUT BETWEEN '2017-01-01' AND '2017-06-01' 
	GROUP BY CARD_NO 
	HAVING COUNT(*)>3 ;

CREATE VIEW V_BOOKS AS
SELECT 	B.BOOK_ID, B.TITLE, BC.NO_OF_COPIES, L.BRANCH_NAME
FROM 	BOOK B, LIBRARY_BRANCH L, BOOK_COPIES BC
WHERE 	B.BOOK_ID=BC.BOOK_ID AND
	BC.BRANCH_ID=L.BRANCH_ID;
