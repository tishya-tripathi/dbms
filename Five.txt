1. 

(SELECT PNO, 
FROM PROJECT P, DEPARTMENT D, EMPLOYEE E
WHERE   P.DNO=D.DNO AND 
        E.SSN=D.MGRSSN AND
        E.LAST_NAME='SCOTT'
) 
UNION (
    SELECT P.PNO FROM
    EMPLOYEE E, WORKS_ON W, PROJECT P
    E.SSN=W.SSN AND
    P.PNO=W.PNO AND
    E.LAST_NAME='SCOTT'
)


2.

SELECT E.SSN, E.NAME, 1.1*E.SALARY 
FROM EMPLOYEE E, PROJECT P, WORKS_ON W
WHERE   P.NAME='IOT' AND
        E.SSN=W.SSN AND 
        P.PNO=W.PNO;



3.

SELECT SUM(E.SALARY), MAX(E.SALARY), MIN(E.SALARY), AVG(E.SALARY)
FROM EMPLOYEE E, DEPARTMENT 
WHERE   E.DNO=D.DNO AND
        D.NAME='ACCOUNTS';

 

4.

SELECT E.USN, E.NAME
FROM EMPLOYEE E2
WHERE   NOT EXISTS ( 
                    SELECT PNO FROM PROJECT
                    WHERE   DNO=5 AND
                            PNO NOT IN (SELECT PNO FROM WORKS_ON W
                                        WHERE W.SSN=E.SSN )
                    );


5.


SELECT D.DNO, D.NAME, COUNT(*) 
FROM EMPLOYEE E, DEPARTMENT D
WHERE   E.DNO=D.DNO AND
        E.SALARY>600000 AND
        E.DNO IN ( SELECT E2.DNO FROM EMPLOYEE E2
        GROUP BY E2.DNO HAVING COUNT(*)>5  )
GROUP BY D.DNO;
